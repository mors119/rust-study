```rust
fn main() {
    println!("1+...+100={}", get_sum(100));
    // !는 매크로로 실행 전 코드로 변경된다.
    let sum = 100;
    // 변수는 중괄호 안에 넣는 것도 가능
    println!("1+00={sum}");
}

fn get_sum(n:u32) -> u32 {
    let mut sum:u32 = 0;

    // 1..=100과 1..101은 동일하다.
    for i in 1..= n {
        sum += i;
    }

    return sum;
}
```
